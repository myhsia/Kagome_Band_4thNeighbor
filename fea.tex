\section{INTRODUCTION TO FEA}

\begin{frame}{The Divide and Conquer Approach}
  \begin{alertblock}{Basic Concept\cite{fea_wiki}}
    \begin{itemize}[<+->]
      \item FEA is a general numerical method for solving partial differential
      equations in two- or three-space variables (i.e., some boundary value
      problems).
      \item It subdivides a large system into smaller, simpler parts called
      \alert{finite elements}.
    \end{itemize}
  \end{alertblock}
  \visible<3->{%
  \begin{block}{From Continuous to Discrete}
    FEA is a numerical strategy to solve complex PDEs by approximating a
    continuous domain $\Omega$ with a collection of simple sub-domains called
    \textbf{elements} ($\Omega_e$).
  \end{block}
  }\pause
  \begin{exampleblock}{Features}
    \vskip-6pt
    \begin{columns}
      \begin{column}{.75\linewidth}
        \begin{itemize}[<+->]
          \item \textbf{Complex Geometry:} Handled by fitting the mesh to the boundary.
          \item \textbf{Local Basis:} Using local ``piecewise'' polynomials
          instead of global functions.
          \item \textbf{Nodal Rep.:} The field is solved only
          at specific points/nodes.
        \end{itemize}
      \end{column}
      \hfill
      \begin{column}{.21\linewidth}

\begin{frame}{Mathematical Principle}
  \begin{block}{From Differential to Integral}
    Instead of solving the Schr\"odinger equation $\mathcal H\psi = E\psi$
    directly at every point, FEA satisfies it \alert{on average} using the
    Method of Weighted Residuals.
  \end{block}
  \pause
  \begin{exampleblock}{Derivation of the Weak Formulation}
    \begin{itemize}[<+->]
      \item \textbf{Residual Definition:} We want $R \to 0$, where
      \[
        R = (\mathcal H - E)\psi.
      \]
      \item \textbf{Weighting (Galerkin):} We multiply $R$ by a test
      function $v$ and integrate over the domain
      \[
        \int_{\Omega} v \ab[-\frac{\hbar^2}{2m} \nabla^2 + V(\bm r) - E]
        \psi \d \Omega = 0.
      \]
      \item \textbf{Integration by Parts:} Lowers the continuity requirement
      on $\psi$
      \[
        \int_{\Omega} \ab[\frac{\hbar^2}{2m} \nabla v \cdot \nabla \psi
        + v V(\bm r) \psi] \d \Omega - \underbrace{
          \oint_{\partial\Omega} \frac{\hbar^2}{2m} v \nabla \psi \cdot
          \mathbf n \d S}_\text{vanishes due to periodicity}
        = E \int_{\Omega} v \psi \d \Omega.
      \]
    \end{itemize}
  \end{exampleblock}
\end{frame}

\section{METHODOLOGY}

\begin{frame}{Calculation Method}
  \visible<1->{%
  \begin{block}{Shape Functions}
    Within an element $\Omega_e$, the unknown field is expanded using local
    basis functions $N_i(\bm r)$
    \begin{equation}
      \psi_e(\bm r) \approx \sum_{i=1}^{n} N_i(\bm r) \psi_i.
    \end{equation}
    }
    \vskip-9pt
    \begin{columns}
      \begin{column}{.72\linewidth}
        \begin{itemize}
          \item<2->
          $\psi_i$ are the values at the nodes.
          \item<3->
          $N_i$ (Shape Functions) are usually linear or
          quadratic polynomials.
          \item<4->
          Typically, $N_i$ has a value of 1 at node $i$
          and 0 at all other nodes.
        \end{itemize}
      \end{column}
      \hfill
      \visible<2->{%
      \begin{column}{.24\linewidth}
        \centering
        \begin{tikzpicture}[scale = .5]
          \draw [thick]
            (0, 0) node [left, font = \scriptsize] {Node 3} --
            (2, 0) node [right, font = \scriptsize] {Node 2} --
            (1, 1.5) -- cycle;
          \draw [->, blue, ultra thick] (1,1.5) -- (1, 2.5)
           node [right] {$N_1=1$};
        \end{tikzpicture}
      \end{column}
      }
    \end{columns}
  \end{block}
  \pause\visible<5->{\only<5>{%
  \begin{exampleblock}{The Element Level}
    For each element, we define the \textbf{Stiffness Matrix} $\mathbf{k}^e$
    and \textbf{Mass Matrix} $\mathbf{m}^e$
    \begin{equation}
      k^e_{ij} = \int_{\Omega_e} \ab(
        \frac{\hbar^2}{2m} \nabla N_i \cdot \nabla N_j + V(\bm r) N_i N_j)
      \d \Omega, \quad m^e_{ij} = \int_{\Omega_e} N_i N_j \d \Omega.
    \end{equation}
    They can be used to construct the global matrices.
  \end{exampleblock}
  }}\visible<6->{%
  \begin{alertblock}{Assembly of Final System}
    The global matrices $\mathbf{K}$ and $\mathbf{M}$ are constructed by
    summing the local matrices, then results in the matrix eigenvalue problem
    \[
      \mathbf{K} = \sum_{e} \mathbf{L}_e^\mathsf{T} \mathbf{k}^e \mathbf{L}_e,
      \quad
      \mathbf{M} = \sum_{e} \mathbf{L}_e^\mathsf{T} \mathbf{m}^e \mathbf{L}_e,
      \quad
      \alert{\mathbf{K} \bm \Psi = E \mathbf{M} \bm \Psi}.
    \]
    where $\mathbf{L}_e$ is the connectivity mapping.
  \end{alertblock}
  }
\end{frame}

\begin{frame}{Element Integration Details}
  \begin{block}{Explicit Shape Functions (Linear Triangle)}
    For a triangular element with nodes at $(x_1,y_1)$, $(x_2,y_2)$,
    and $(x_3,y_3)$, the shape functions are barycentric coordinates
    \[
      N_i(x,y) = \frac{a_i + b_i x + c_i y}{2\Delta},
    \]
    where $\Delta$ is the triangle area and the coefficients are given by
    cyclic permutations of coordinates.
  \end{block}
  \pause
  \begin{exampleblock}{Analytical Integration}
    For linear triangles, the gradients are constants: $\nabla N_i
    = \frac{1}{2\Delta} \begin{psmallmatrix} b_i \\ c_i \end{psmallmatrix}$.
    Hence the stiffness contribution becomes
    \[
      k^e_{ij} \approx \frac{\hbar^2}{8m\Delta^2} (b_i b_j + c_i c_j)
               + V(\bar{\bm r}) \int_{\Omega_e} N_i N_j \d \Omega,
    \]
    with the mass matrix entry
    \[
      \int_{\Omega_e} N_i N_j \d \Omega = \frac{\Delta}{12} (1 + \delta_{ij}).
    \]
  \end{exampleblock}
\end{frame}

\begin{frame}{Transition: From Tight-Binding to FEA}
  \begin{columns}
    \begin{column}{.56\linewidth}
      \begin{block}{Horizontal Comparison of TB and FEA}
        \scriptsize \centering
        \begin{tabular}{>{\bfseries}lp{8.5em}p{11em}}
          \toprule
          Methods     & Tight-Binding & Finite Element Analysis       \\
          \midrule
          Character   & Discrete      & Continuous                    \\
          Basis       & Atomic orbitals at different sites
                      & Piecewise polynomials on a mesh               \\
          Interaction & Hopping integrals $t_{ij}$
                      & Potential field $V(\bm r)$                    \\
          Hamiltonian & $3 \times 3$ matrix
                      & Large sparse matrices $\mathbf K$, $\mathbf M$\\
          \bottomrule
        \end{tabular}
      \end{block}
    \end{column}
    \pause
    \begin{column}{.4\linewidth}
      \begin{alertblock}{Governing Equation}\small
        Time-independent Schr\"o\-dinger equation
        \[
          \ab[-\frac{\hbar^2}{2m} \nabla^2 + V(\bm r)] \psi(\bm r)
        = E \psi(\bm r),
        \]
        where $V(\bm r)$ is the periodic potential of the Kagome lattice.
      \end{alertblock}
    \end{column}
  \end{columns}
  \pause
  \begin{exampleblock}{The Numerical Workflow}
    \begin{enumerate}
      \item<3-> \textbf{Geometry:} Define the Kagome unit cell (three
      potential wells).
      \item<4-> \textbf{Mesh:} Generate a fine triangular mesh inside the cell.
      \item<5-> \textbf{Scan:} Choose a path in the Brillouin Zone
      (e.g., $\Gamma \to K \to M \to \Gamma$).
      \item<5-> \textbf{Solve:} For each $\bm k$, solve the generalized
      eigenvalue problem for the first few $E_n(\bm k)$.
      \item<5-> \textbf{Plot:} Connect the eigenvalues to form the
      dispersion curves.
    \end{enumerate}
  \end{exampleblock}
\end{frame}

\begin{frame}{Energy Bands via Eigenvalue Analysis}
  \begin{block}{Imposing Periodicity}
    In a crystal lattice, the wavefunction must satisfy \alert{Bloch's Theorem}
    \[
      \psi(\bm r + \bm a_n) = \upe^{\iu \bm k \cdot \bm a_n} \psi(\bm r).
    \]
  \end{block}
  \pause
  \begin{columns}
    \begin{column}{.51\linewidth}
      \begin{exampleblock}{Transform to FEA Representation}
        \begin{itemize}[<+->]
          \item In FEA, this is implemented as a constraint on the boundary
          nodes of the unit cell mesh.
          \item The nodes on the ``left'' boundary are related to those on
          the ``right'' via the factor $\upe^{\iu \bm k \cdot \bm a}$.
          \item This makes the global matrix $\mathbf{K}$ dependent on the
          crystal momentum (wavevector) $\bm k$
          \[
            \mathbf K(\bm k) \bm \Psi_{\bm k}
          = E(\bm k) \mathbf M \bm \Psi_{\bm k},
            \quad
            \bm k = (k_x, k_y, k_z).
          \]
          \vskip-6pt
        \end{itemize}
      \end{exampleblock}
    \end{column}
    \visible<5->{%
    \begin{column}{.45\linewidth}
      \begin{alertblock}{Comparing Results}
        \begin{itemize}[<+->]
          \item \textbf{Tight-Binding:} Predicts a flat band and two
          dispersive bands.
          \item \textbf{FEA}
          \begin{enumerate}[<+->]
            \item  Recovers the same structure in the low-energy limit.
            \item But also shows \textbf{higher energy bands} (excited states)
            not captured by simple tight-binding models.
          \end{enumerate}
        \end{itemize}
      \end{alertblock}
    \end{column}
    }
  \end{columns}
\end{frame}

\begin{frame}{Implementation of Bloch Periodic Boundary Conditions}
  \begin{block}{Node Classification}
    The unit cell boundary nodes are partitioned into:
    \pause
    \begin{itemize}[<+->]
      \item \textbf{Master nodes} on the left/bottom edges.
      \item \textbf{Slave nodes} on the right/top edges
      (related to masters by $\bm a_1$, $\bm a_2$).
      \item \textbf{Interior nodes} free of constraints.
    \end{itemize}
  \end{block}
  \visible<5->{%
  \begin{exampleblock}{Matrix Condensation}
    Let $\bm \Psi = (\bm \Psi_I, \bm \Psi_M, \bm \Psi_S)^\mathsf{T}$.
    The constraint is
    \[
      \bm \Psi_S = \mathbf{T}(\bm k) \bm \Psi_M, \quad
      \mathbf T(\bm k) = \operatorname{diag}(\upe^{\iu\bm k\cdot\bm a_1},
      \ldots, \upe^{\iu\bm k\cdot\bm a_2}, \ldots).
    \]
    Substituting into $\mathbf K \bm \Psi = E \mathbf M \bm \Psi$ yields a
    reduced generalized eigenvalue problem
    \[
      \mathbf K_\text{red}(\bm k) \tilde{\bm \Psi}
      = E \mathbf M_{\text{red}}(\bm k) \tilde{\bm \Psi},
      \]
      where $\tilde{\bm \Psi} = (\bm \Psi_I, \bm \Psi_M)^\mathsf{T}$ and
      the reduced matrices are Hermitian for all $\bm k$.
    \end{exampleblock}
  }
\end{frame}

  us to study the effect of the \textbf{physical shape} of the lattice sites

    simulate vacancies.
